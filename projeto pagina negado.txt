negado.js

// ...

document.addEventListener('DOMContentLoaded', function() {
    showPhraseSequentially();
});
const textContainer = document.getElementById('text');
const phrases = [
    "-Seu cartão foi negado.",
    "É, vamos precisar te levar de volta para a terra.",
    "Porém você não poderá pagar essa passagem pelo visto né",
    "Bem, verei com o meu comandante o que podemos fazer..."
];

let currentPhraseIndex = 0;

function showPhraseSequentially(){
    let textContainer = document.getElementById('text'); // Mova esta linha para dentro da função.

    if (currentPhraseIndex >= phrases.length){
        return;
    }

    textContainer.textContent = phrases[currentPhraseIndex];
    currentPhraseIndex++;

    setTimeout(showPhraseSequentially, 3000);
}

const videoContainer = document.getElementById('video-container');
const videoPath = 'moon.mp4'; // Caminho para o vídeo

const video = document.createElement('video');
video.autoplay = true;
video.muted = false; // Altere para "false" para iniciar o vídeo com som.

// Verifica se o vídeo pode ser carregado corretamente
video.oncanplay = () => {
    if (!videoStarted) {
        showPhraseIndex();
        videoStarted = true;
    }
};

// Trata erros de carregamento do vídeo
video.onerror = () => {
    console.error('Erro ao carregar o vídeo.');
};

video.src = videoPath;

videoContainer.appendChild(video);
